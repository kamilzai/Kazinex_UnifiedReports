# User UI Specification

**Date:** December 11, 2025  
**Purpose:** User interface for report viewing, editing, and management  
**Users:** All authenticated users (Editors and Viewers)

---

## Overview

The user UI provides:
- **Report Library** - Browse and search reports
- **Report Viewer/Editor** - View and edit report data in grid
- **Multi-Section Support** - Tabs for different sections
- **Excel-like Experience** - Copy/paste, keyboard navigation, inline editing
- **Image Upload** - Drag and drop images with preview
- **Copy from Previous** - Wizard to copy data from previous report
- **Batch Operations** - Add/delete multiple rows

**UI Framework:** Ant Design 5.x + AG Grid Community  
**Layout:** Header + main content  
**URL Structure:** `/reports/*`

// src/components/reports/CellEditors/NumberEditor.tsx

export const NumberEditor = forwardRef((props: ICellEditorParams, ref) => {
  const [value, setValue] = useState(props.value);
  const inputRef = useRef<HTMLInputElement>(null);

  useImperativeHandle(ref, () => ({
    getValue: () => {
      // Return parsed number
      return parseFloat(value) || 0;
    },
  }));

  useEffect(() => {
    // Focus input
    inputRef.current?.focus();
    inputRef.current?.select();
  }, []);

  return (
    <input
      ref={inputRef}
      type="number"
      value={value}
      onChange={(e) => setValue(e.target.value)}
      onKeyDown={(e) => {
        if (e.key === 'Enter') {
          props.stopEditing();
        }
      }}
      style={{
        width: '100%',
        height: '100%',
        border: 'none',
        padding: '8px',
        fontSize: '14px',
      }}
    />
  );
});
